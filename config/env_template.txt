# PostgreSQL SQL Analyzer - Конфигурация LLM
# Скопируйте этот файл в .env и заполните своими API ключами

# =============================================================================
# ОБЩИЕ НАСТРОЙКI AI
# =============================================================================

# Включить AI-рекомендации (true/false)
SQL_ANALYZER_ENABLE_AI_RECOMMENDATIONS=true

# Провайдер AI (auto, openai, anthropic, local)
SQL_ANALYZER_AI_PROVIDER=auto

# Порог уверенности для рекомендаций (0.0-1.0)
SQL_ANALYZER_AI_CONFIDENCE_THRESHOLD=0.7

# =============================================================================
# OPENAI НАСТРОЙКИ
# =============================================================================

# OpenAI API ключ (получите на https://platform.openai.com/api-keys)
OPENAI_API_KEY=sk-your-openai-api-key-here

# Модель OpenAI (gpt-4, gpt-3.5-turbo, gpt-4-turbo)
SQL_ANALYZER_OPENAI_MODEL=gpt-4

# Температура для генерации (0.0-2.0)
SQL_ANALYZER_OPENAI_TEMPERATURE=0.7

# =============================================================================
# ANTHROPIC НАСТРОЙКИ
# =============================================================================

# Anthropic API ключ (получите на https://console.anthropic.com/)
ANTHROPIC_API_KEY=sk-ant-your-anthropic-api-key-here

# Модель Anthropic (claude-3-sonnet, claude-3-haiku, claude-3-opus)
SQL_ANALYZER_ANTHROPIC_MODEL=claude-3-sonnet

# =============================================================================
# ЛОКАЛЬНЫЕ МОДЕЛИ
# =============================================================================

# URL локального LLM сервера (например, Ollama)
SQL_ANALYZER_LOCAL_LLM_URL=http://localhost:11434

# Модель для локального LLM
SQL_ANALYZER_LOCAL_LLM_MODEL=llama2

# =============================================================================
# ДОПОЛНИТЕЛЬНЫЕ НАСТРОЙКИ
# =============================================================================


# Пороги для анализа
SQL_ANALYZER_LARGE_TABLE_THRESHOLD=1000000
SQL_ANALYZER_EXPENSIVE_QUERY_THRESHOLD=1000.0
SQL_ANALYZER_SLOW_QUERY_THRESHOLD=100.0

# =============================================================================
# ИНСТРУКЦИИ ПО НАСТРОЙКЕ
# =============================================================================

# 4. Для тестирования используйте тестовую базу данных
